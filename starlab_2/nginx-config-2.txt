server {
    listen 80;
    listen [::]:80;
    server_name bsuir.stacklevel.group;

    location / {
        return 308 https://$host$request_uri;
    }
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name bsuir.stacklevel.group;

    ssl_certificate     /etc/letsencrypt/live/bsuir.stacklevel.group/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/bsuir.stacklevel.group/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    location / {
        proxy_pass http://localhost:8081;
        proxy_http_version 1.1;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
    
    location ^~ /admin-parol/ {
        proxy_pass http://localhost:8083;
        proxy_http_version 1.1;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_set_header Upgrade           $http_upgrade;
        proxy_set_header Connection        "upgrade";

        proxy_read_timeout 300s;
        proxy_send_timeout 300s;
    }

    location ^~ /summernote/ {
        proxy_pass http://localhost:8083;
        proxy_http_version 1.1;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 300s;
        proxy_send_timeout 300s;
    }
    
    location ^~ /media/ {
        proxy_pass http://localhost:8082;
        proxy_set_header Host $host;
        access_log off;
    }
    
    location ^~ /static/ {
        proxy_pass http://localhost:8083;
        proxy_set_header Host $host;
        access_log off;
    }
}
